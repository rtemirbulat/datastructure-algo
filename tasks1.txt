1. 3x3 matrix with numbers: 
3 2 1 
2 3 2 
1 2 3



#include <iostream>
#include <malloc.h>

using namespace std;
int main()
{
	const int numTwo = 2;
	const int numThree = 1;
	int n;
	cin >> n; 
	int** arr;
	arr = new int* [n];
	for (int i = 0; i < n; i++)
		arr[i] = new int[n];
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			if (i == j) arr[i][j] = n;
			else if ((i - j == 1) || (j - 1 == 1)) arr[i][j] = numTwo;
			else if ((i - j == 2) || (j - i == 2)) arr[i][j] = numThree;
		}
	}

	//print 
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			cout << arr[i][j] << " ";
		}
		cout << endl;
	}

	//delete array

}
